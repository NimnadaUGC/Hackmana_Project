CREATE DATABASE hakmanaEdm;

USE hakmanaEdm;

CREATE TABLE User (
    userNIC VARCHAR(15) PRIMARY KEY,
    name VARCHAR(20),
    title VARCHAR(20),
    gmail VARCHAR(25),
    verification_code CHAR(6) NULL,
    code_expiry DATETIME NULL
);

CREATE TABLE Desktop (
    regNum VARCHAR(20) PRIMARY KEY,
    model VARCHAR(25),
    status VARCHAR(10),
    serialNum VARCHAR(30),
    purchasedFrom VARCHAR(25),
    ram VARCHAR(8),
    processor VARCHAR(10),
    warranty VARCHAR(25),
    hardDisk VARCHAR(8),
    os VARCHAR(10),
    floppyDisk VARCHAR(3),
    soundCard VARCHAR(15),
    tvCard VARCHAR(15),
    networkCard VARCHAR(15),
    monitorRegNum VARCHAR(20),
    projectorRegNum VARCHAR(20),
    speakerRegNum VARCHAR(20),
    mouseRegNum VARCHAR(20),
    keyboardRegNum VARCHAR(20),
    micRegNum VARCHAR(20),
	scannerRegNum VARCHAR(20),
    userNIC VARCHAR(15),
    FOREIGN KEY (userNIC) REFERENCES User (userNIC)
);

CREATE TABLE Printer (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    serialNum VARCHAR(30),
    paperInput VARCHAR(15),
    paperOutput VARCHAR(15),
    warranty VARCHAR(25),
    userNIC VARCHAR(15),
    FOREIGN KEY (userNIC) REFERENCES User (userNIC)
);

CREATE TABLE Laptop (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    RAM VARCHAR(20),
    CPU VARCHAR(100),
    Storage VARCHAR(100),
    Display VARCHAR(100),
    GraphicsCard VARCHAR(100),
    OperatingSystem VARCHAR(100),
    userNIC VARCHAR(15),
    FOREIGN KEY (userNIC) REFERENCES User (userNIC)
);

CREATE TABLE PhotoCopyMachine (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    CopyingCapability VARCHAR(20)
);

CREATE TABLE Monitors (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    userNIC VARCHAR(15),
    regNumDesktop VARCHAR(20),
    FOREIGN KEY (userNIC) REFERENCES User (userNIC),
    FOREIGN KEY (regNumDesktop) REFERENCES Desktop (regNum)
);

CREATE TABLE MultimediaProjector (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10)
);


CREATE TABLE FaxMachine (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10)
);

CREATE TABLE UPS (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    BackupPower VARCHAR(50),
    Runtime VARCHAR(50),
    regNumDesktop VARCHAR(20),
    FOREIGN KEY (regNumDesktop) REFERENCES desktop (regNum)
);


CREATE TABLE WebCam (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    regNumDesktop VARCHAR(20),
    FOREIGN KEY (regNumDesktop) REFERENCES desktop (regNum)
);


CREATE TABLE Tripod (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    Stability VARCHAR(50),
    AdjustableHeight VARCHAR(50)
);

CREATE TABLE Headphone (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    SoundQuality VARCHAR(50),
    Connectivity VARCHAR(50)
);

CREATE TABLE USBHub (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10)
);

CREATE TABLE ExternalDVDRom (
     regNum VARCHAR(20) PRIMARY KEY NOT NULL,
     model VARCHAR(25),
     status VARCHAR(10)
);

CREATE TABLE AccessPoint (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10)
);

CREATE TABLE ExternalUSB (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    usbType VARCHAR(10)
);

CREATE TABLE SpeakerSet (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    SoundQuality VARCHAR(50),
    numberOfSpeakers INT
);

CREATE TABLE Router (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    numberOfPorts INT
);

CREATE TABLE ToolKit (
     regNum VARCHAR(20) PRIMARY KEY NOT NULL,
     model VARCHAR(25),
     status VARCHAR(10)
);

CREATE TABLE USBCable (
     regNum VARCHAR(20) PRIMARY KEY NOT NULL,
     model VARCHAR(25),
     status VARCHAR(10),
     length DOUBLE(5,2)
);

CREATE TABLE RouterSwitch (
     regNum VARCHAR(20) PRIMARY KEY NOT NULL,
     model VARCHAR(25),
     status VARCHAR(10)
);

CREATE TABLE VGACable (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10)
);

CREATE TABLE Keyboard (
     regNum VARCHAR(20) PRIMARY KEY NOT NULL,
     model VARCHAR(25),
     status VARCHAR(10),
    Connectivity VARCHAR(50)
);

CREATE TABLE Mouse (
    regNum VARCHAR(20) PRIMARY KEY NOT NULL,
    model VARCHAR(25),
    status VARCHAR(10),
    Connectivity VARCHAR(50)
);

CREATE TABLE systemUser (
    userName VARCHAR(50),
    fullName VARCHAR(100),
    post VARCHAR(50),
    empId VARCHAR(50),
    pwd VARCHAR(64),
    email VARCHAR(50),
    phoneNum VARCHAR(10),
    isRemember BOOLEAN DEFAULT FALSE,
    PRIMARY KEY (userName)
);


CREATE TABLE notes(
    regNumDevice varchar(20) NULL,
    username varchar(20) NULL,
    model varchar(20) NULL,
    notes varchar(60),
    PRIMARY KEY(regNumDevice,username,model,notes)
);
